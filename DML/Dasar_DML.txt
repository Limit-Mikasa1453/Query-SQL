DML atau Data Manipulation Language merupakan perintah dasar SQL 
yang bertujuan untuk memanipulasi data yang ada dalam suatu database

-INSERT
  INSERT INTO table_name (column1, column2, column3, ...)
  VALUES (value1, value2, value3, ...);
  
  INSERT INTO table_name
  VALUES (value1, value2, value3, ...);

-SELECT 
  SELECT column1, column2, ...
  FROM table_name;
  
-UPDATE
  UPDATE table_name
  SET column1 = value1, column2 = value2, ...
  WHERE condition;
  
-DELETE
  DELETE FROM table_name WHERE condition;
